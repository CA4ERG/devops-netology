#include <stdio.h> // библиотека ввода-вывода
#include <stdlib.h> // стандартная библиотека функций общего назначения
#include <locale.h> // библиотека локализации (в программе devcppPortable выводит непонятные символы)




/* run this program using the console pauser or add your own getch, system("pause") or input loop */


#include <stdio.h>

int main() {
	setlocale(LC_ALL, "Russian"); // устанавливаем локализацию на вывод - русский язык
	
	
	double massiv[3][3]; // Объявление массива 3x3
	int i, j, x, k; //x будет осуществлять выбор какую матрицу будем использовать

	

	printf("Какую матрицу вы хотите использовать\n1)Матрица 3x3 - цифра 1\n2)Матрица 2x2 - цифра 2\n\n");
	scanf("%d", &x);

	if(x==1) {	
	// Ввод элементов массива с клавиатуры
	printf("Введите элементы массива 3x3 (9 цифр через пробел):\n");
	for (i = 0; i < 3; i++) { // Внешний цикл по строкам
		for (j = 0; j < 3; j++) { // Внутренний цикл по столбцам
			//printf("Введите элемент [%d][%d]: ", i, j); //- лишнее но пока оставить
			scanf("%lf", &massiv[i][j]); // Считываем число, & - оператор показывает куда записываем число в массиве
		}
	}

	// Вывод элементов массива на экран
	printf("\nВаш массив 3x3:\n");
	for (i = 0; i < 3; i++) {
		for (j = 0; j < 3; j++) {
			printf("%8.2lf ", massiv[i][j]); // Выводим элемент %8.2lf - 8 длина поля (с учетом точки и цифр после неё) .2 (сколько цифр после точки, lf - тип данных long float
		}
		printf("\n"); // Переход на новую строку после каждой строки массива
	}


	//сумма элементов на основной и побочной диагонали массива 3 на 3
	
	double summaosnova3na3 = massiv[0][0] + massiv[1][1] + massiv[2][2];
	printf("сумма главной диагонали будет равно = %8.2lf\n", summaosnova3na3);
	double summadopoln3na3 = massiv[2][0] + massiv[1][1] + massiv[0][2];
	printf("сумма дополнительной диагонали будет равно = %8.2lf\n", summadopoln3na3);
	
}
	

	else if (x == 2) {

	int massivik[2][2]; // Объявление массива 2x2

	// Ввод элементов массива с клавиатуры
	printf("Введите элементы массива 2x2 (4 цифры через пробел):\n");
	for (i = 0; i < 2; i++) { // Внешний цикл по строкам
		for (j = 0; j < 2; j++) { // Внутренний цикл по столбцам
			//printf("Введите элемент [%d][%d]: ", i, j); //- лишнее но пока оставить
			scanf("%d", &massivik[i][j]); // Считываем число & - оператор показывает куда записываем число в массиве
		}
	}

	// Вывод элементов массива на экран
	printf("\nВаш массив 2x2:\n");
	for (i = 0; i < 2; i++) {
		for (j = 0; j < 2; j++) {
			printf("%d ", massivik[i][j]); // Выводим элемент
		}
		printf("\n"); // Переход на новую строку после каждой строки массива
	}

	//Для вывода квадрата массива 2 на 2 добавлю ещё элемент q и массив квадрата данной матрицы quadro [][] и решу по формуле quadro [i][j] += massivik[i][k] * [massivik [k][j]
	
	double quadro[2][2];
	int k;
	
	for (i=0; i<2; i++) {
		for (j=0; j<2; j++) {
			for (k=0; k<2; k++) {
				quadro[i][j] += massivik[i][k] * massivik[k][j]; // += это сложение с дальнейшим присваиванием данному элементу (В нашем случае quadro) значения
			}
		}
	}
	
	printf("\nКвадрат массива 2x2:\n");
	for (i = 0; i < 2; i++) {
		for (j = 0; j < 2; j++) {
			printf("%8.0lf ", quadro[i][j]); // Выводим элемент
		}
		printf("\n"); // Переход на новую строку после каждой строки массива
	}
}

		     else {
        printf("Неверный ввод. Пожалуйста, введите 1, 2 или 3.\n");
    }

	return 0;
}
