#include <stdio.h> // библиотека ввода-вывода
#include <stdlib.h> // стандартная библиотека функций общего назначения
#include <locale.h> // библиотека локализации (в программе devcppPortable выводит непонятные символы)


int main() {
	setlocale(LC_ALL, "Russian");
	
	FILE *f1;
	FILE *f2;
	char s[1024]; //В интернете указано, что размер строки 1024 достаточно для текстового файла
	f1=fopen("fio_goda.txt","rt");
	f2=fopen("ludy_bolshe_1980.txt","w");
	
	while (fgets(s,sizeof(s), f1)) {
		
		char *posledniy = strrchr(s, ' '); //функция strrchr ищет последний пробел, поскольку наш текстовый файл предоставлен в строгом виде (ФИО год)
		if (s != NULL) { // если строка прочитана, мы продолжаем читать дальше
			int got = atoi(posledniy + 1); // сдвигаем указатель posledniy на один символ (чтобы он пропустил последний пробел и указывал на первую цифрук года рождения), также с помощью atoi конвертируем эту часть строки в целое число
			
			if (got > 1980 && got <=3000) { // сравниваем полученные результаты и выкидываем лишнее
				fprintf(f2, "%s", s); // записываем в наш файл f2
			}
		}
	}
	
	printf("Строки, к которым относятся люди, родившиеся позднее 1980 года, были успешно записаны в файл 'ludy_bolshe_1980.txt'\n %s", f2);
	
	
	fclose(f1);
	fclose(f2);
	
	return 0;
}
